81 uint32_t HX711_read(void)
82 {
83  // wait for the chip to become ready
84  while (!HX711_isReady());
85 
86  unsigned long count;
87  unsigned char i;
88 
89  GPIOOn(internal_dout);//DOUT_SET_HIGH;
90 
91  DelayUs(1);
92 
93  GPIOOff(internal_pd_sck);//PD_SCK_SET_LOW;
94  DelayUs(1);
95 
96  count=0;
97  while(GPIORead(internal_dout));
98  for(i=0;i<24;i++)
99  {
100  GPIOOn(internal_pd_sck);//PD_SCK_SET_HIGH;
101  DelayUs(1);
102  count=count<<1;
103  GPIOOff(internal_pd_sck);//PD_SCK_SET_LOW;
104  DelayUs(1);
105  if(GPIORead(internal_dout))
106  count++;
107  }
108  count = count>>6;
109  GPIOOn(internal_pd_sck);//PD_SCK_SET_HIGH;
110  DelayUs(1);
111  GPIOOff(internal_pd_sck);//PD_SCK_SET_LOW;
112  DelayUs(1);
113  count ^= 0x800000;
114  return(count);
115 }
